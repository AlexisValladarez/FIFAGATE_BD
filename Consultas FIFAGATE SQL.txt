SELECT NOMBRE, APELLIDO, NACIONALIDAD
FROM PERSONAS 
ORDER BY NOMBRE;

SELECT NOMBRE, APELLIDO, NACIONALIDAD
FROM PERSONAS 
WHERE NACIONALIDAD LIKE '_r%';

SELECT PERSONASID, NOMBRE, APELLIDO, NACIONALIDAD
FROM PERSONAS 
WHERE PERSONASID BETWEEN 'P021' AND 'P037';

SELECT PERSONASID, NOMBRE, NACIONALIDAD
FROM PERSONAS 
WHERE PERSONASID = 'P027';

SELECT PERSONASID, NOMBRE, APELLIDO, NACIONALIDAD
FROM PERSONAS 
WHERE PERSONASID IN ('P015', 'P023', 'P030');

SELECT NOMBRE, APELLIDO, NACIONALIDAD
FROM PERSONAS 
WHERE NACIONALIDAD = 'Ecuador';

SELECT NOMBRE, APELLIDO, NACIONALIDAD
FROM PERSONAS 
WHERE APELLIDO LIKE 'A%' OR NACIONALIDAD LIKE '_o%';

SELECT P.NOMBRE, P.APELLIDO, S.TIPO_SENTENCIA, P.NACIONALIDAD
FROM PERSONAS P, SENTENCIA S
WHERE P.PERSONASID = S.PERSONAID;

SELECT P.NOMBRE, P.APELLIDO, S.TIPO_SENTENCIA, P.NACIONALIDAD
FROM PERSONAS P, SENTENCIA S
WHERE P.PERSONASID = S.PERSONAID
AND S.TIPO_SENTENCIA LIKE '%Arrestado%';

SELECT P.NOMBRE, P.APELLIDO, S.TIPO_SENTENCIA, PA.OCUPACION
FROM PERSONAS P, SENTENCIA S, PERSONAS_ARRESTADAS PA
WHERE P.PERSONASID = PA.PERSONAID AND P.PERSONASID=S.PERSONAID;

SELECT P.NOMBRE, P.APELLIDO, P.NACIONALIDAD, S.TIPO_SENTENCIA, CI.OCUPACION
FROM PERSONAS P, SENTENCIA S, CAPTURA_INTERNACIONAL CI
WHERE P.PERSONASID = CI.PERSONAID AND P.PERSONASID=S.PERSONAID
AND S.TIPO_SENTENCIA LIKE '%Orden de captura internacional%';

SELECT P.NOMBRE, P.APELLIDO, P.NACIONALIDAD, S.TIPO_SENTENCIA, CI.OCUPACION
FROM PERSONAS P, SENTENCIA S, CAPTURA_INTERNACIONAL CI
WHERE P.PERSONASID = CI.PERSONAID AND P.PERSONASID=S.PERSONAID
AND S.TIPO_SENTENCIA LIKE '%Orden de captura internacional%'
AND P.NACIONALIDAD LIKE '%Brasil%';

SELECT P.NOMBRE, P.APELLIDO, PS.OCUPACION, PS.FECHA
FROM PERSONAS P, PERSONAS_SUSPENDIDAS PS
WHERE P.PERSONASID = PS.PERSONAID;

SELECT P.NOMBRE, P.APELLIDO, PS.OCUPACION, PS.FECHA
FROM PERSONAS P, PERSONAS_SUSPENDIDAS PS
WHERE P.PERSONASID = PS.PERSONAID
AND PS.FECHA > '01/10/2015';

SELECT P.NOMBRE ||' '|| P.APELLIDO AS "CAMBIOS EN ALTOS CARGOS" , P.NACIONALIDAD,CAC.FUNCIONARIO_ENTRANTE, CAC.FUNCIONARIO_SALIENTE
FROM PERSONAS P, CAMBIOS_ALTOS_CARGOS CAC
WHERE P.PERSONASID = CAC.FUNCIONARIO_ENTRANTE
ORDER BY NACIONALIDAD;

SELECT P.NOMBRE ||' '|| P.APELLIDO AS "PERSONAS SUSPENDIDAS" , P.NACIONALIDAD
FROM PERSONAS P, PERSONAS_SUSPENDIDAS PS
WHERE P.PERSONASID = PS.PERSONAID;

SELECT P.NOMBRE ||' '|| P.APELLIDO AS "PERSONAS CAPTURA INTERNACIONAL" , P.NACIONALIDAD
FROM PERSONAS P, CAPTURA_INTERNACIONAL CI
WHERE P.PERSONASID = CI.PERSONAID;

SELECT P.NOMBRE ||' '|| P.APELLIDO AS "PERSONAS ARRESTADAS" , P.NACIONALIDAD, S.TIPO_SENTENCIA
FROM PERSONAS P, PERSONAS_ARRESTADAS PA, SENTENCIA S
WHERE P.PERSONASID = PA.PERSONAID AND P.PERSONASID = S.PERSONAID;

SELECT P.NOMBRE ||' '|| P.APELLIDO AS "PERSONAS ARRESTADAS" , P.NACIONALIDAD, S.TIPO_SENTENCIA
FROM PERSONAS P, PERSONAS_ARRESTADAS PA, SENTENCIA S
WHERE P.PERSONASID = PA.PERSONAID AND P.PERSONASID = S.PERSONAID 
AND S.SENTENCIAID = 'S025';

SELECT P.NOMBRE, P.APELLIDO, P.NACIONALIDAD
FROM PERSONAS P, SENTENCIA S
WHERE P.NACIONALIDAD = 'Argentina' AND P.PERSONASID = S.PERSONAID;
